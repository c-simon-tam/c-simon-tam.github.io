function updateClock() {
    const now = new Date();
    const hours = String(now.getHours()).padStart(2, '0');
    const minutes = String(now.getMinutes()).padStart(2, '0');
    const seconds = String(now.getSeconds()).padStart(2, '0');
    
    document.getElementById('digitalTime').textContent = 
        `${hours}:${minutes}:${seconds}`;
    
    updateBinaryColumn('h1', parseInt(hours[0]));
    updateBinaryColumn('h2', parseInt(hours[1]) << 1);
    updateBinaryColumn('m1', parseInt(minutes[0]));
    updateBinaryColumn('m2', parseInt(minutes[1]) + 1);
    updateBinaryColumn('s1', parseInt(seconds[1]));
    updateBinaryColumn('s2', parseInt(seconds[0]) * 2);
}

function updateBinaryColumn(columnId, digit) {
    let binary = digit.toString(2).padStart(4, '0');
    
    // Special case for M1 column
    if (columnId === 'm1') {
        binary = binary.split('').reverse().join('');
    }
    
    const bits = [8, 4, 2, 1];
    bits.forEach((value, index) => {
        const bitElement = document.getElementById(`${columnId}-${value}`);
        const isOn = binary[index] === '1';
        
        bitElement.textContent = binary[index];
        bitElement.className = isOn ? 'bit on' : 'bit off';
    });
    
    document.getElementById(`${columnId}-decimal`).textContent = digit;
}
